def &(other)
  raise ArgumentError.new("Expected at least two expression on stack, got #{@stack.size}") if @stack.size < 2
  right = @stack.pop
  left = @stack.pop
  expr = Expression.new_complete(left, :&, right)
  @stack.push expr
  self
end
